function minAndMax(l, d, x) {
  let result = [];
  let n;
  let m;

  for (let i = l; i <= d; i++) {
    let digits = i.toString().split("");
    let digitSum = digits.reduce((acc, x) => acc + Number(x), 0);
    
    if (digitSum == x) {
      n = Number(i);
      break;
    };
  }
  result.push(n);

  for (let i = d; i >= l; i--) {
    let digits = i.toString().split("");
    let digitSum = digits.reduce((acc, x) => acc + Number(x), 0);
    if (digitSum == x) {
      m = Number(i);
      break;
    };
  }
  result.push(m);

  return result;
}

Magic sum of 3s - CW_DONE

function magicSum(numbers) {
  let result = 0;
  numbers.forEach((x) => {
    if ( x.toString().includes("3") && x % 2 != 0) result += x;
  });
  return result;
}
